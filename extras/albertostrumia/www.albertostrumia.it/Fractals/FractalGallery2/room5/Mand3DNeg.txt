/************************************************	Mand3DNeg.c**  Generates a RAW format file of a 3D image *  of Mandelbrot set (grayscale)**  C source file by Alberto Strumia*************************************************/#include<stdio.h>/* definition of constants */#define Radius 10#define Side 0.7	#define Cx -1.5*Side	#define Cy 0.9*Side#define M 700				#define Num 50		#define scale 2		#define co 1.41		#define si 1.41			/* main program */main(){		int p, q, n, w, r[M];				double x, y, xx, yy, Incx, Incy;		FILE *fp;	fp = fopen("Mand.raw","w");			for (p = 1; p <= M; p++)	{					printf("%i %%\n", p*100/M);					for (q = 1; q <= M; q++)		{			Incx = Cx - Side + 2*Side*(p*co+q*si)/M;			Incy = Cy - Side + 2*Side*(-p*si+q*co)/M;					x = 0;		y = 0;		w = 0;				for ( n = 1; n <= Num; ++n)			{				xx = x*x - y*y - Incx;				yy = 2*x*y - Incy;								x = xx;				y = yy;								if ( x*x + y*y > Radius )					{						w = n;						n = Num;					}			}							for ( n = 1; n <= w; n++)			{				if ( q > n/scale )					{						r[q + n/scale] = w;					}							}		}				for ( q = 1; q <= M; q++)			{						fprintf(fp, "%c", r[q] );				r[q] = 255;				}	}		fclose(fp);	}/* end of main program */
