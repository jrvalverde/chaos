/************************************************	Mand3Dint.c**  Generates a RAW format file of a 3D image *  of Mandelbrot set (grayscale)**  C source file by Alberto Strumia*************************************************/#include<stdio.h>/* definition of constants */#define Radius 10#define Side 1.2#define Cx -1.2*Side#define Cy 2.1*Side#define scale 5#define M 700#define Num 256#define co 1.41#define si 1.41/* main program */main(){		int p, q, n, wext, wint, r[M];			double x, y, xx, yy, Incx, Incy, md;			FILE *fp;	fp = fopen("Mand.raw","w");			for (p = 1; p <= M; p++)	{					printf("%i %%\n", p*100/M);				for (q = 1; q <= M; q++)		{			Incx = Cx - Side + 2*Side*(p*co+q*si/2)/M;			Incy = Cy - Side + 2*Side*(-p*si+q*co/2)/M;					x = 0;		y = 0;			for ( n = 1; n <= Num; ++n)			{				xx = x*x - y*y - Incx;				yy = 2*x*y - Incy;								x = xx;				y = yy;				md = x*x + y*y;				wint = md*Num/Side/Side;				wext = wint;				r[q] = wint;				if ( md > Radius )				  {					wext = n;					n = Num;				  }			}				for ( n = 1; n <= wext; n++)			{				if ( q > n/scale )				   {					r[q + n/scale] = wext;				   }			}		}						for ( q = 1; q <= M; q++)			{						fprintf(fp, "%c", r[q] );			}	}		fclose(fp);	/* end of main program */
